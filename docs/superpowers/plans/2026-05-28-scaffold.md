# elvy-map Scaffold Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Stand up a runnable full-stack skeleton (React+TS frontend, FastAPI backend, multi-service dev container) with an OpenAPI-typed frontend client and a thin end-to-end slice.

**Architecture:** Monorepo. FastAPI backend exposes typed Pydantic responses → OpenAPI schema. The schema is exported to `backend/openapi.json` and `openapi-typescript` generates `frontend/src/lib/api-schema.d.ts`, consumed by an `openapi-fetch` client. A MapTiler map + geocoding control selects a coordinate; the frontend calls the backend for (stub) metrics. Docker Compose runs both services as the dev container.

**Tech Stack:** Python 3.12, FastAPI, uv, SQLModel (SQLite), pytest, ruff. React 18 + TypeScript, Vite, MapTiler SDK JS, openapi-fetch/openapi-typescript, eslint. Docker Compose dev container.

**Spec:** `docs/superpowers/specs/2026-05-27-elvy-map-scaffold-design.md`

### Task 10: MapView component (MapTiler SDK + geocoding)

Render a MapTiler map centered on Sweden, with a geocoding control that yields the picked coordinate via `onSelect`. Shows a clear message if the API key is missing.

**Docs to check:** MapTiler geocoding-control README (https://docs.maptiler.com/sdk-js/modules/geocoding/) for the `pick` event shape — code below reads `event.detail.center` as `[lon, lat]`; adjust if the installed version differs.

**Files:**
- Create: `frontend/src/components/MapView.tsx`

- [ ] **Step 1: Write `frontend/src/components/MapView.tsx`**

```tsx
import { useEffect, useRef } from "react";
import * as maptilersdk from "@maptiler/sdk";
import "@maptiler/sdk/dist/maptiler-sdk.css";
import { GeocodingControl } from "@maptiler/geocoding-control/maptilersdk";
import "@maptiler/geocoding-control/style.css";

const apiKey = import.meta.env.VITE_MAPTILER_KEY as string | undefined;

type Props = {
  onSelect?: (lat: number, lon: number) => void;
};

export function MapView({ onSelect }: Props) {
  const containerRef = useRef<HTMLDivElement | null>(null);
  const onSelectRef = useRef(onSelect);
  onSelectRef.current = onSelect;

  useEffect(() => {
    const container = containerRef.current;
    if (!container || !apiKey) return;

    maptilersdk.config.apiKey = apiKey;
    const map = new maptilersdk.Map({
      container,
      style: maptilersdk.MapStyle.STREETS,
      center: [15.0, 62.0], // Sweden
      zoom: 4,
    });

    const gc = new GeocodingControl({ apiKey });
    map.addControl(gc);

    const onPick = (event: Event) => {
      const feature = (event as CustomEvent).detail;
      const coords: [number, number] | undefined = feature?.center;
      if (coords) onSelectRef.current?.(coords[1], coords[0]);
    };
    gc.addEventListener("pick", onPick);

    return () => {
      gc.removeEventListener("pick", onPick);
      map.remove();
    };
  }, []);

  if (!apiKey) {
    return (
      <div style={{ padding: 16, fontFamily: "sans-serif" }}>
        Set <code>VITE_MAPTILER_KEY</code> in <code>frontend/.env</code> to load
        the map.
      </div>
    );
  }

  return <div ref={containerRef} style={{ width: "100%", height: "100%" }} />;
}
```

```tsx
import { useEffect, useState } from "react";

import { MapView } from "./components/MapView";
import { getHealth, getMetrics } from "./lib/api";

type Metrics = Awaited<ReturnType<typeof getMetrics>>;

export default function App() {
  const [backendOk, setBackendOk] = useState<boolean | null>(null);
  const [metrics, setMetrics] = useState<Metrics | null>(null);

  useEffect(() => {
    getHealth()
      .then(() => setBackendOk(true))
      .catch(() => setBackendOk(false));
  }, []);

  async function handleSelect(lat: number, lon: number) {
    try {
      setMetrics(await getMetrics(lat, lon));
    } catch {
      setMetrics(null);
    }
  }

  return (
    <div style={{ display: "flex", height: "100vh" }}>
      <div style={{ flex: 1 }}>
        <MapView onSelect={handleSelect} />
      </div>
      <aside style={{ width: 320, padding: 16, fontFamily: "sans-serif" }}>
        <h1>elvy-map</h1>
        <p>
          Backend:{" "}
          {backendOk === null ? "checking…" : backendOk ? "ok" : "down"}
        </p>
        {metrics ? (
          <ul>
            <li>lat: {metrics.lat}</li>
            <li>lon: {metrics.lon}</li>
            <li>cloud cover: {metrics.cloud_cover_pct}%</li>
            <li>lightning prob: {metrics.lightning_probability}</li>
            <li>note: {metrics.note}</li>
          </ul>
        ) : (
          <p>Search an address to see stub metrics.</p>
        )}
      </aside>
    </div>
  );
}
```

up:
	$(COMPOSE) up --build

down:
	$(COMPOSE) down
